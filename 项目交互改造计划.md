### 0.1 RAF Delta 缓冲

#### [MODIFY] [useStudioBridge.ts](file:///D:/visual_ProgrammingSoftware/毕设and简历Projects/Xnova-Code/apps/studio/src/renderer/hooks/useStudioBridge.ts)

**位置**：L890 `bridge.runtime.onEvent` 回调中的 `text_delta` 和 `thinking` 分支（L1009-L1020, L1131-L1266）

**当前问题**：每个 `text_delta` 直接调用 `setLiveConversation`（L1131），触发整棵 React 树 reconciliation。

**修改方案**：在 hook 顶部创建一个 RAF 缓冲器，将高频 delta 累积后每帧 flush 一次。

```typescript
// 在 useStudioBridge 内部，bridge useEffect 之前添加：
const pendingTextDeltaRef = useRef('')
const pendingThinkingDeltaRef = useRef('')
const rafIdRef = useRef<number | null>(null)

const flushPendingDeltas = useCallback(() => {
  rafIdRef.current = null
  const textChunk = pendingTextDeltaRef.current
  const thinkingChunk = pendingThinkingDeltaRef.current
  pendingTextDeltaRef.current = ''
  pendingThinkingDeltaRef.current = ''

  if (textChunk) {
    setLiveConversation((current) =>
      appendLiveTextBlock(
        finalizeOpenThinkingBlocks(current),
        createLiveBlockId('text'),
        textChunk,
      ),
    )
  }
  if (thinkingChunk) {
    setLiveConversation((current) =>
      appendLiveThinkingBlock(current, createLiveBlockId('thinking'), thinkingChunk),
    )
  }
}, [])

const scheduleFlush = useCallback(() => {
  if (rafIdRef.current === null) {
    rafIdRef.current = requestAnimationFrame(flushPendingDeltas)
  }
}, [flushPendingDeltas])
```

然后将 L1131 的 `setLiveConversation` 中 `text_delta` 和 `thinking` case 改为累积：

```typescript
case 'text_delta':
  pendingTextDeltaRef.current += typeof event.payload?.text === 'string'
    ? event.payload.text : ''
  scheduleFlush()
  return  // 不再直接 setLiveConversation

case 'thinking':
  pendingThinkingDeltaRef.current += typeof event.payload?.text === 'string'
    ? event.payload.text : ''
  scheduleFlush()
  return  // 不再直接 setLiveConversation
```

### 1.1 移植 Batcher

#### [NEW] `apps/studio/src/main/adaptive-event-batcher.ts`

**核心参考文件**：[OC/src/main/ipc/adaptive-event-batcher.ts](file:///D:/visual_ProgrammingSoftware/毕设and简历Projects/OpenCowork-main/src/main/ipc/adaptive-event-batcher.ts)

需要适配的关键差异：

| OC 事件类型 | Xnova 事件类型 | 适配方式 |
|------------|--------------|---------|
| `text_delta` (AgentStreamEvent) | `text_delta` (StudioRuntimeEvent) | payload.text 累积 |
| `thinking_delta` | `thinking` | payload.text 累积 |
| `tool_use_args_delta` | 无对应 | 跳过 |
| `loop_end` / `error` | `run_completed` / `run_failed` / `run_cancelled` | 终端事件立即 flush |

**核心结构**（简化自 OC L22-L48, L54-L222）：

```typescript
import type { StudioRuntimeEvent, StudioRuntimeEventType } from '../shared/studio-bridge-contract'

interface BatcherConfig {
  foregroundFlushMs: number   // 33ms — 前台可见时
  backgroundFlushMs: number   // 150ms — 后台窗口时
  maxBufferSize: number       // 200 字符上限触发强制 flush
}

const AGGREGATABLE_TYPES: Set<StudioRuntimeEventType> = new Set([
  'text_delta', 'thinking',
])

const TERMINAL_TYPES: Set<StudioRuntimeEventType> = new Set([
  'run_completed', 'run_failed', 'run_cancelled',
  'turn_end', 'session_end', 'model_request_failed',
])

interface RunAccumulator {
  runId: string
  textDelta: string
  thinkingDelta: string
  timer: ReturnType<typeof setTimeout> | null
}

export class AdaptiveEventBatcher {
  private config: BatcherConfig
  private runs = new Map<string, RunAccumulator>()
  private isForeground = true
  private handler: ((event: StudioRuntimeEvent) => void) | null = null

  constructor(config?: Partial<BatcherConfig>) {
    this.config = {
      foregroundFlushMs: 33,
      backgroundFlushMs: 150,
      maxBufferSize: 200,
      ...config,
    }
  }

  setHandler(handler: (event: StudioRuntimeEvent) => void): void {
    this.handler = handler
  }

  setForeground(visible: boolean): void {
    this.isForeground = visible
  }

  push(event: StudioRuntimeEvent): void {
    const runId = event.runId ?? '__default__'

    if (AGGREGATABLE_TYPES.has(event.type)) {
      let acc = this.runs.get(runId)
      if (!acc) {
        acc = { runId, textDelta: '', thinkingDelta: '', timer: null }
        this.runs.set(runId, acc)
      }
      this.accumulate(acc, event)
      this.scheduleFlush(acc)
      return
    }

    // 控制事件：先 flush 累积 delta，再立即发送
    this.flushRun(runId)
    this.handler?.(event)

    if (TERMINAL_TYPES.has(event.type)) {
      this.cleanupRun(runId)
    }
  }

  // ... accumulate / flushRun / scheduleFlush / cleanupRun / stop
}
```

import remarkGfm from 'remark-gfm'

export function MarkdownContent({ text, className }: MarkdownContentProps) {
  return (
    <div className={`md-content ${className ?? ''}`}>
      <Markdown remarkPlugins={[remarkGfm]}
        components={{
          code: ({ children, className: codeClass, ...props }) => {
            const isInline = !codeClass
            if (isInline) {
              return <code className="md-inline-code" {...props}>{children}</code>
            }
            return <code className={codeClass} {...props}>{children}</code>
          },
          a: ({ href, children, ...props }) => {
            if (href && /^https?:\/\//i.test(href)) {
              return <a href={href} target="_blank" rel="noreferrer noopener" className="md-link" {...props}>{children}</a>
            }
            return <span>{children}</span>
          },
        }}
      >
        {text}
      </Markdown>
    </div>
  )
}
